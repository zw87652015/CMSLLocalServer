CMSLLocalServer/
│
├── app.py                     # Flask主应用入口，包含所有路由和API端点
├── tasks.py                   # Celery任务定义，包含COMSOL执行逻辑和队列管理
├── models.py                  # SQLAlchemy数据库模型定义
├── config.py                  # 应用配置（路径、密钥、限制等）
├── database.db                # SQLite数据库文件（存储用户、任务信息）
│
├── requirements.txt           # Python依赖列表
├── .env                       # 环境变量配置（敏感信息）
│
├── uploads/                   # 用户上传的原始.mph文件存储
│   └── Filename.mph
│
├── results/                   # COMSOL处理后的结果文件存储
│   └── Filename_solved.mph
│
├── logs/                      # 系统运行日志和错误日志
│   └── comsol_20230831.log
│
├── templates/                 # HTML模板文件
│   ├── index.html             # 主界面，文件上传和任务监控
│   ├── history.html           # 用户历史记录查看页面
│   └── queue.html             # 全局任务队列状态页面
│
└── static/                    # 静态资源
    ├── style.css              # 统一界面样式
    ├── script.js              # 客户端交互逻辑
    └── favicon.ico            # 网站图标
